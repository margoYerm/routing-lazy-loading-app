{{ str }} //❴❴ str ❵❵

{{ str | uppercase }} //❴❴ str | uppercase ❵❵

{{ str | uppercase | lowercase }} //❴❴ str | uppercase | lowercase ❵❵

{{ str | titlecase }} //❴❴ str | titlecase ❵❵

{{ str | titlecase | slice:1 }} //❴❴ str | titlecase | slice:1 ❵❵

{{ str | titlecase | slice:1:5 }} //❴❴ str | titlecase | slice:1:5 ❵❵

{{ str | titlecase | slice:1:5-2 }} //❴❴ str | titlecase | slice:1:5-2 ❵❵
